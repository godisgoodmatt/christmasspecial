#include intrographics_justladyluck.txt
#include mariahcareybiggraphics.txt

setdatapath("data/graphics/cutscenes/thief/");

var thiefgraphics = {
	idle: {
		animation: "thief_idle.json",
		pack: "thief.pck",
		loop: true
	},
	worried_idle: {
		animation: "thief_worried_idle.json",
		pack: "thief.pck",
		loop: true
	},
	transform: {
		animation: "thief_transform.json",
		pack: "thief.pck",
		speed: 2,
		loop: false
	}
}
var thief = addsprite("thief", 1, thiefgraphics);

// HELPER FUNCTIONS

var fade_time = 0.4;

// ACTORS
actor("Mariah Carey", screenwidthmid + 800, 550, Col.YELLOW, CENTER, BOTTOM, "chat_mariahcarey", "characters/misc/charicon_mariahcarey");
actor("Thief", 2600, 780, Col.multiplylightness(Col.GREEN, 1.25), CENTER, BOTTOM, "chat_thief", "characters/thief/charicon_thief");

// INTRO

play("music_cutscene_dark");

thief.scale = 0.5;
thief.x = 2620 - ((thief.getwidth("idle") * thief.scale) / 2);
thief.y = 890;
changebackground("intro_stage");
fadein();
thief.show("worried_idle");

mariahcarey_big.x = -screenwidth;
mariahcarey_big.show("idle");
Actuate.tween(mariahcarey_big, 2, { x: 0 }).ease(Expo.easeOut);

wait(0.5);

mariahcarey_big.show("talking", true);
actor("Mariah Carey", screenwidthmid + 350, 700, Col.YELLOW, CENTER, BOTTOM, "chat_mariahcarey", "characters/misc/charicon_mariahcarey");
speak("Mariah Carey", "Hi there, thief!");
speak("Mariah Carey", "We have something special lined up for you today.");
speak("Thief", "For some reason I'm not sure I'm going to enjoy it.");
mariahcarey_big.show("talking-sly", true);
speak("Mariah Carey", "Nonsense my friend.");
speak("Mariah Carey", "I'm sure it'll give you a whole new perspective on things!");
speak("Mariah Carey", "Now, stay exactly where you are.");
speak("Thief", "Uh... okay.");

fadeout();
while(isfading()) {
	// do nothing
}
changebackground("intro_transform");
thief.hide();
mariahcarey_big.hide();
fadein();

thief.x = screenwidth;
thief.y = 0;
thief.scale = 1;
var transform = thief.show("transform");
Actuate.tween(thief, 2, { x: 0 }).ease(Expo.easeOut);

var marker_human = false;
var marker_dice = false;
while(true) {
	if(!marker_human && transform.marker == "human") {
		marker_human = true;
		transform.paused = true;
        speak("Thief", "Wait, why is the spotlight on?");
        speak("Thief", "What's going on?");
		play("transformintodice");
		transform.paused = false;
	} else if(!marker_dice && transform.marker == "dice") {
		marker_dice = true;
		transform.paused = true;
        actor("Bear", 2600, 780, Col.WHITE, CENTER, BOTTOM, null, "characters/bear/charicon_bear");
		speak("Bear", "[70%]RRRRAWWRR RAWR RRRAWR.");
        speak("Mariah Carey", "You're not supposed to be that size...");
        speak("Mariah Carey", "Don't worry, I'll get Alchemist to make you a growth potion.|We'll get you back to the right size in just a second...");
	  break;
	}
}

startgamefromcutscene();
